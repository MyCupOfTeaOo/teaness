## 上传动画

<Playground>
  {
    () => {
      const [count, setCount] = useState(0)
      const [state, setState] = useState([{
        name: "test.txt",
        percent: 20,
        uid: "1",
        status: "updating"
      },{
        name: "test2.txt",
        percent: 0,
        uid: "2",
        status: "updating"
      }])
      useEffect(() => {
        setState([{
          name: "test.txt",
          percent: 0,
          uid: "1",
          status: "updating"
        },{
          name: "test2.txt",
          percent: 0,
          uid: "2",
          status: "updating"
        }])
        let isCancel = false;
        function go() {
          setTimeout(() => {
            setState(prev => {
              if (isCancel) return
              prev.forEach(item => {
                const diff = Math.floor(Math.random() * 10)
                Object.assign(item, {
                  ...item,
                  percent: Math.min(item.percent + diff,100)
                })
              })
              if (prev.some(item => item.percent < 100)) {
                go()
              }
              return [...prev]
            })
          },Math.floor(Math.random() * 1000))
        }
        const time = setTimeout(go,200);
        () => {
          clearTimeout(time);
          isCancel = true;
        }
      },[count])
      return (
        <div style={{
          padding: 24,
          backgroundColor: "rgba(0, 0, 0, 0.5)"
        }}>
          <Button onClick={() => setCount(prev => prev + 1)}>refresh</Button>
          <UploadProgress progresses={state}/>
        </div>
      )
    }
  }
</Playground>
