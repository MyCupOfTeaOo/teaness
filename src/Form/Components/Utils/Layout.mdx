# maskLayout

> 一个布局工具类,帮助简单合并布局

## 基本用法(Basic usage)

export const layout = maskLayout(
  horizontal,
  {
    label: {
      float: 'left',
      labelStyle: {
        minWidth: 100,
        paddingRight: 24,
      },
    },
  },
  { depth: true },
);

<div style={{ padding: 24 }}>
  <Layout layout={layout}>
    <Label text="姓名1">
      <Input />
    </Label>
    <Label text="姓名2">
      <Input />
    </Label>
    <Label text="姓名3">
      <Input />
    </Label>
  </Layout>
</div>

## 参数模版

<Playground>
  {() => {
    // 该代码应该定义在外面,此处只是为了演示把该步骤放在了useMemo里
    const useLayout = useMemo(() =>  maskLayout(
      horizontal,
      (float) => ({
        label: {
          float,
          labelStyle: {
            minWidth: 100,
            paddingRight: 24,
          },
        },
      }),
      { depth: true },
    ))
    const [float, setFloat] = useState("left")

    const layout = useLayout(float);
    return (
      <div>
        <Radio.Group buttonStyle="solid" value={float} onChange={(e) => setFloat(e.target.value)}>
          <Radio.Button value="left">
            left
          </Radio.Button>
          <Radio.Button value="right">
            right
          </Radio.Button>
        </Radio.Group>
        <Layout layout={layout}>
          <Label text="姓名1">
            <Input />
          </Label>
          <Label text="姓名2">
            <Input />
          </Label>
          <Label text="姓名3">
            <Input />
          </Label>
        </Layout>
      </div>
    );
  }}
</Playground>
